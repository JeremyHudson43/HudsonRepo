create table Employee (
SSN int(8) NOT NULL UNIQUE,
dob	date NOT NULL,
Fname varchar(20) NOT NULL,
Minit varchar(1) NOT NULL,
Lname varchar(20) NOT NULL, 
Address varchar(150),
primary key (SSN));

create table Dependent (
Name char(20) NOT NULL,
Relationship varchar(15) NOT NULL,
SSN	int(8) NOT NULL UNIQUE,
primary key (SSN, Relationship, Name),
foreign key (SSN) references Employee(SSN));


create table Project (
ProjName varchar(15) NOT NULL,
ProjNum integer(15) NOT NULL UNIQUE,
ProjDesc varchar(60) NOT NULL, 
primary key (ProjNum, ProjName));

create table SalariedEmp (
monthlySal int(20) NOT NULL,
SSN	int(8) NOT NULL UNIQUE,
primary key (SSN),
foreign key (SSN) references Employee(SSN));

create table Department (
deptNum	integer(25) NOT NULL UNIQUE,
deptName varchar(50) NOT NULL,
managerSSN int(8) NOT NULL,
primary key (deptNum), 
foreign key (managerSSN) references salariedEmp(SSN));

create table DeptLocation (
deptNum	integer(25) NOT NULL UNIQUE,
Location varchar(25) NOT NULL,
primary key (deptNum, Location),
foreign key (deptNum) references Department(deptNum));

create table Works (
SSN int(8) NOT NULL UNIQUE,
ProjName varchar(15) NOT NULL,
ProjNum  integer(15) NOT NULL UNIQUE,
deptNum integer(25) NOT NULL,
primary key (SSN, ProjName, ProjNum, deptNum),
foreign key (SSN) references Employee(SSN),
foreign key (deptNum) references department(deptNum),
foreign key (ProjNum, ProjName) references Project(ProjNum, ProjName));

create table HourlyEmp (
hourlyRate integer(10) NOT NULL,
SSN int(8) NOT NULL UNIQUE,
primary key (SSN),
foreign key (SSN) references Employee(SSN));
